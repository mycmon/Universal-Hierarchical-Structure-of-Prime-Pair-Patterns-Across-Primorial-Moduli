```python
def generate_patterns(modulus, diffs):
    residues = residues_coprime_to_modulus(modulus)
    patterns_by_k = {}
    for k in diffs:
        patterns = set()
        for r in residues:
            rk = (r + k) % modulus
            if gcd(rk, modulus) == 1:
                patterns.add((residue_to_label(r), residue_to_label(rk)))
        patterns_by_k[k] = patterns
    return patterns_by_k
```
